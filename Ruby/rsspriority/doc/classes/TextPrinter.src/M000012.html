        def get_item_output(i)
                super(i)

                s = "Title: " + i.get_title() + "\n\n\n"
                s += "Source: " + i.get_channel_desc() + "\n\n"
                s += "Desc: " + i.get_content(strip_html_tags=1) + "\n" 
                s += "Url: " + i.get_url() + "\n"
                s += "Score: " + i.get_score().to_s() + "\n"
                s += "\n\n\n"

                return s
        end
